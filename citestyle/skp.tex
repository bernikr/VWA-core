% Quote blocks
% \begin{(long)quote}[Seitenzahl]{Quelle}Text\end{(long)quote}
\newcommand*{\TempSource}{} % Helper function as parameters of env. can only be used in the first codeblock
\renewenvironment*{quote}[2][]
{%
	\renewcommand*{\TempSource}{\parencite[][#1]{#2}}%
	\glqq%
}{%
	\grqq\enspace
	\TempSource
}
\newenvironment*{longquote}[2][]
{
	\renewcommand*{\TempSource}{\parencite[][#1]{#2}}
	\vspace{0.5cm}
	\begin{addmargin}{1cm}
}
{
	\TempSource
	\end{addmargin}
	\vspace{0.5cm}
}
